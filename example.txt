    string palabras[2] = {"hola", "mundo"};
    string saludo = "adios";
    bool flags[2] = {true, false};
    bool result;

    for (int i = 0; i < 3; i++) {
        printf("%d\n", nums[i]);
    }
    printf("%s\n", palabras[1]);

    if (flags[0]) {
        printf("%s\n", "Bandera activa\n");
    }

    printf("%s\n", saludo);

    fun();

    result = fun();

    return 0;
}
